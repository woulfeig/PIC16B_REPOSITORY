#importing our function from a seperate python file saved within our folder
from climate_database import query_climate_database
import inspect

def query_climate_database(db_file, country, year_begin, year_end, month):

    """
    Function that uses SQL commands to organize the data from our three CSV files into the created columns

    Arguments:  db_file - the file for our database (in the form of a string)
                country - string that describes the name of the country 
                year_begin - integer value that states the start year 
                year_end - integer value that states the end year 
                month - integer value that states the specific month of date we are working with 
    
    Returns: df - the result is a pandas data frame that organizes the values from the arguments explained above
    """

    conn = sqlite3.connect(db_file)
    query = f'''
        SELECT s.NAME, s.LATITUDE, s.LONGITUDE, c.NAME AS Country, t.Year, t.Month, t.Temp
        FROM temperatures t
        LEFT JOIN stations s ON t.ID = s.id
        LEFT JOIN countries c ON t.ID LIKE c."FIPS 10-4" || '%'
        WHERE c.Name = '{country}' AND t.Month = {month} AND t.Year BETWEEN {year_begin} AND {year_end}

'''
    result_df = pd.read_sql_query(query, conn)
    conn.close()
    return result_df

#our function will allow us to create out plot
def temperature_coefficient_plot(db_file, country, year_begin, year_end, month, min_obs, **kwargs):
    """
    Function will create a scatter map box plot that explores the average yearly change in temperature 

    Arguments: db_file - the file for our database that we will use to make our plot (in the form of a string)
                country - string that describes the name of the country that will appear on our plot
                year_begin - integer value that states the start year for our plot 
                year_end - integer value that states the end year for our plot
                month - integer value that states the specific month of date we plotting

    Returns: fig - a scatterplot map box that shows the map of a country with the different 
                   measurements placed on the location where they were taken 
    """
    df = query_climate_database(db_file, country, year_begin, year_end, month)
    df = df[df.groupby('NAME')['Year'].transform('count') >= min_obs]

    #average tenperature will allow us to use one measurement per station 
    def temperature_average(data):
        """
        Function will be used to determine the average temperature recorded in a station during a given year

        Arguments: data['Year'] - the integer value from our dataframe that looks at a single year 
                                 in between our year_begin and our year_end
                   data['Temp'] - the intger value that describes the temperature recorded in the 
                                  given year

        Returns: average - an integer value that will be used to calculate the yearly change
        """

        X = data['Year']
        y = data['Temp']
        
        average = np.polyfit(X, y, deg=1)[0]
        
        return average

    #creates columns for our visalization
    coefs = df.groupby('NAME').apply(temperature_average).reset_index()
    coefs.columns = ['NAME', 'YearlyChange']

    df = pd.merge(df, coefs, on='NAME')

    #this type of visualization will appear as an interactive map with data points
    fig = px.scatter_mapbox(df, 
                            lat='LATITUDE', 
                            lon='LONGITUDE',
                            color='YearlyChange',
                            hover_data = {'NAME' : True, 'YearlyChange': ':.3f'},
                            labels = {'YearlyChange': 'Estimated Yearly Increase (Celsius)'}, 
                            title = f"Estimates of Yearly Increase in Temperature in {pd.to_datetime(month, format='%m').month_name()} for stations in {country}, {year_begin}-{year_end}",
                            **kwargs)
    fig.update_layout(mapbox_style="open-street-map")
    return fig

#uses a scatterplot to explore the temperature changes within a given country
def regional_difference_plot(db_file, country, year_begin, year_end, month, min_obs, **kwargs):
    """
    Function will create a scatter map box plot that explores how temperatures differ in North 
    Italy compared to South Italy

    Arguments: db_file - the file for our database that we will use to make our plot (in the form of a string)
                country - string that describes the name of the country that will appear on our plot
                year_begin - integer value that states the start year for our plot 
                year_end - integer value that states the end year for our plot
                month - integer value that states the specific month of date we plotting

    Returns: fig - a scatterplot map box that shows the map of a country with the different 
                   measurements placed on the location where they were taken that will make it very 
                   easy to compare the differences between regions of one country
    """
    df = query_climate_database(db_file, country, year_begin, year_end, month)
    df = df[df.groupby('NAME')['Temp'].transform('count') >= min_obs]

Multi-Faceted Graphes

Multi-faceted graphes are a great way for us to share more information through out graphes. Facets allow us to break up our graphs so that we have a figure that resembles two plots side by side. For our data set, graphs with facets are great for us to show examine the varying temperatures of two different countries. This leads us to our question, who experienced warmer temperatures in August of 2000: Brazil or Mexico?

Let’s start by defininng our function! We will be using data from our second query function.

def month_vs_temp(db_file, country1, country2, year, month, **kwargs): 
    """
    Function will create a basic scatter plot that explores how temperatures differ in Brazil 
    and Mexico during the given month and year

    Arguments: db_file - the file for our database that we will use to make our plot (in the form of a string)
                country - string that describes the name of the country that will appear on our plot
                year - integer value that describes the year the data was collected
                month - integer value that states the specific month of date we plotting

    Returns: fig - a scatterplot that will plot Month vs Temperature for two countries contained in 
                   seperate facets that will make it appear as two graphes plotted on one plane
    """

      #using our second query function with four inputs
    df1 = second_query_climate_database(db_file, country1, year, month)
    df2 = second_query_climate_database(db_file, country2, year, month)

      #defining an equation for our coefficients
    def coef(group):
          X = group['Year']
          y = group['Temp']
          coef = np.polyfit(X, y, deg=1)[0]
          return coef

      #using our coefficients to combine our two countries into one graph
    coefs1 = df1.groupby('NAME').apply(coef).reset_index()
    coefs2 = df2.groupby('NAME').apply(coef).reset_index()
    coefs1.columns = ['NAME','YearlyChange']
    coefs1['Country'] = country1
    coefs2.columns = ['NAME', 'YearlyChange']
    coefs2['Country'] = country2
   
    coefs_combined = pd.concat([coefs1, coefs2])
    df1 = pd.merge(df1, coefs1, on='NAME') 
    df2 = pd.merge(df2, coefs2, on='NAME')
    df_combined = pd.concat([df1, df2])
    
      #defining our scatterplot function
    fig = px.scatter(df_combined, x='Month',
                     y='Temp',
                     hover_data={'NAME': True, 'Month': True, 'Temp' : True},
                     labels={'Month': 'Month', 'Temp': 'Temperature (°C)'},
                     title=f'Differences in Temperatures within a Year for {country1} and {country2} during August', 
                     facet_col='Country_x',
                     **kwargs)
    
    fig.update_xaxes(range=[1, 12])
    fig.update_yaxes(range=[-10, 40])

    return fig
